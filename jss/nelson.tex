
\section{Nelson/Siegel and Svensson method}
\label{sec:nels-svenss-meth}


\begin{equation}
  \label{eq:nelsonspot}
   s(m,\bm{b}) = \beta_0 + \beta_1\frac{1-\exp(-\frac{m}{\tau_1})}{\frac{m}{\tau_1}} + \beta_2\left(\frac{1-\exp(-\frac{m}{\tau_1})}{\frac{m}{\tau_1}} - \exp(-\frac{m}{\tau_1})\right)
\end{equation}
